### 5.2 驗證代碼

```python
def rolling_backtest_big_lotto(num_bets):
    all_draws = db_manager.get_all_draws('BIG_LOTTO')
    hits = 0
    
    for target_idx, target_draw in enumerate(all_draws):
        if target_idx == 0:
            continue
        
        # 關鍵: 使用該期 **之後** 的全部歷史
        history = all_draws[target_idx + 1:]
        
        # 基於歷史預測
        predictions = []
        for bet_idx in range(num_bets):
            pred = predict_with_method(
                history,
                method='zone_balance',
                window=500 if bet_idx % 2 == 0 else 200
            )
            predictions.append(pred)
        
        # 檢查命中
        actual_numbers = target_draw['numbers']
        for pred in predictions:
            if len(set(pred) & set(actual_numbers)) >= 3:
                hits += 1
                break
    
    return hits / len(all_draws)
```

## 附件 B: 技術細節

### zone_balance 方法的工作原理

```python
def zone_balance(history, window=500):
    """
    區間平衡策略:
    1. 分割號碼成 6 個區間 (1-8, 9-16, ..., 41-49)
    2. 統計每個區間的熱度
    3. 從熱度高的區間選號
    """
    zones = [
        (1, 8), (9, 16), (17, 24),
        (25, 32), (33, 40), (41, 49)
    ]
    
    # 統計窗口內的區間分佈
    zone_freq = {z: 0 for z in zones}
    for draw in history[:window]:
        for num in draw['numbers']:
            for zone_idx, (start, end) in enumerate(zones):
                if start <= num <= end:
                    zone_freq[zone_idx] += 1
    
    # 從高頻區間選號
    selected = []
    for zone_idx in sorted(zone_freq, key=lambda x: zone_freq[x], reverse=True)[:3]:
        start, end = zones[zone_idx]
        zone_nums = [n for n in history[0]['numbers'] 
                     if start <= n <= end]
        selected.extend(zone_nums[:2])
    
    return selected[:6]
```
