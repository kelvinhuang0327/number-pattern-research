```python
# 核心回測邏輯
def rolling_backtest_multibet(lottery_type, num_bets, window_sizes):
    """
    多注策略滾動式回測
    
    Args:
        lottery_type: 樂透類型 (POWER_LOTTO)
        num_bets: 注數 (2, 4, 6, 8)
        window_sizes: 每注的窗口大小列表
    
    Returns:
        中獎統計字典
    """
    all_draws = db_manager.get_all_draws(lottery_type)
    hits = 0
    total = 0
    
    for i, target_draw in enumerate(all_draws):
        if i == 0:
            continue
        
        history = all_draws[i:]  # 該期後的全部歷史
        
        # 生成多注預測
        predictions = []
        for bet_idx in range(num_bets):
            window = window_sizes[bet_idx]
            pred = predict_with_window(history, window)
            predictions.append(pred)
        
        # 檢查任意注是否中獎
        actual = target_draw['numbers']
        for pred in predictions:
            if len(set(pred) & set(actual)) >= 3:
                hits += 1
                break
        
        total += 1
    
    return {
        'win_rate': hits / total,
        'hits': hits,
        'total': total,
        'expected_period': total / hits if hits > 0 else 0
    }
```
