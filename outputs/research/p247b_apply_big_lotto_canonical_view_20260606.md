CREATE VIEW IF NOT EXISTS draws_big_lotto_canonical_main AS
SELECT d.*
FROM draws d
WHERE d.lottery_type = 'BIG_LOTTO'
  AND d.draw NOT LIKE '%-%'
  AND NOT (LENGTH(d.draw) = 8 AND d.draw LIKE '20%')
  AND (
    SELECT MAX(CAST(j.value AS INTEGER))
    FROM json_each(d.numbers) j
  ) > 25
